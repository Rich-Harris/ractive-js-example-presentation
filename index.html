});

Data Updates


var buses = { a: 1, b: 2 };
var ractiveView = new Ractive({
  el: $el,
  template: template,
  data: {
    stop: busStop,
    buses: buses
  }
});
buses.c = 4;
ractiveView.update('buses');
					

var buses = [];
var count = 0;
var ractiveView = new Ractive({
  el: $el,
  template: template,
  data: {
    stop: busStop,
    buses: buses
  }
});

onBusUpdate(function(data) {
  buses.push(data[count++]);
});
